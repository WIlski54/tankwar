# Tank Wars v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A playable third-person Three.js tank that drives, shoots, and has visibly running tracks (spinning wheels + scrolling belt tread, with differential steering) built from a Meshy single-mesh GLB.

**Architecture:** A one-time Python **build step** parses the merged Meshy GLB and splits it into named nodes (`Hull`, `Turret`, `Barrel`, `WheelsLeft/Right`, `BeltLeft/Right`), giving the belts fresh arc-length UVs and a `TrackBelt` material → `assets/tank.glb`. The **runtime** (`index.html` + ES modules) loads that GLB, binds the parts, animates wheels/belt from a pure drive model, aims a turret, fires shells, and renders a third-person scene.

**Tech Stack:** Python 3.14 + numpy (build, dependency-light: GLB parsed/written by hand). Three.js r0.16x via CDN import-map (runtime). Tests: plain-assert Python scripts + node's built-in `node --test`.

**Reference spec:** `docs/superpowers/specs/2026-06-28-tank-wars-v0.1-design.md`

**Known geometry facts (from spec probe):** bbox X ±0.95 (length), Y ±0.42 (up), Z ±0.49 (width; tracks at ±Z). Running-gear band ≈ `Y < 0.04 & |Z| > 0.30`, ~126k verts/side. Front direction and turret/wheel/belt thresholds are confirmed empirically in Task 3.

### Task 7 — ADDENDUM: collision helpers in `game/core.js`
Append these pure functions to `game/core.js` (after the existing exports):

```js
// --- arena collision (pure, XZ plane) ---
// wall: { x, z, hw, hd } = center + half-extents on X and Z. Tanks/shells use XZ only.
export function resolveWalls(p, r, walls) {
  let x = p.x, z = p.z;
  for (const w of walls) {
    const nx = Math.max(w.x - w.hw, Math.min(x, w.x + w.hw));
    const nz = Math.max(w.z - w.hd, Math.min(z, w.z + w.hd));
    let dx = x - nx, dz = z - nz;
    const d2 = dx*dx + dz*dz;
    if (d2 < r*r) {
      if (d2 > 1e-12) {                    // outside the box, overlapping a face/corner
        const d = Math.sqrt(d2), push = (r - d) / d;
        x += dx * push; z += dz * push;
      } else {                              // center inside the box: eject least-penetration axis
        const ox = w.hw + r - Math.abs(x - w.x);
        const oz = w.hd + r - Math.abs(z - w.z);
        if (ox < oz) x += (x >= w.x ? ox : -ox);
        else         z += (z >= w.z ? oz : -oz);
      }
    }
  }
  return { x, z };
}
export function pointInWall(p, w, pad = 0) {
  return Math.abs(p.x - w.x) <= w.hw + pad && Math.abs(p.z - w.z) <= w.hd + pad;
}
export function anyWallHit(p, walls, pad = 0) {
  for (const w of walls) if (pointInWall(p, w, pad)) return true;
  return false;
}
```

```js
// game/arena.js — Tron-style neon arena: dark glossy ground, neon grid, maze walls.
import * as THREE from "three";

export const ARENA = 160;   // arena is ARENA x ARENA, centered at the origin
export const WALL_H = 7;    // wall height in world units
export const NEON = 0x18e0ff;

// Open-maze layout: 4 border walls + interior cover. Each {x,z,hw,hd} is an XZ box.
export const WALLS = (() => {
  const b = ARENA / 2, t = 1.2;
  return [
    { x: 0,  z:  b, hw: b, hd: t }, { x: 0,  z: -b, hw: b, hd: t },
    { x:  b, z: 0,  hw: t, hd: b }, { x: -b, z: 0,  hw: t, hd: b },
    { x: -40, z: -20, hw: 1.2, hd: 16 },
    { x: -10, z:  30, hw: 18,  hd: 1.2 },
    { x:  24, z: -8,  hw: 1.2, hd: 20 },
    { x:  44, z:  34, hw: 14,  hd: 1.2 },
    { x:  6,  z: -38, hw: 22,  hd: 1.2 },
    { x: -52, z:  44, hw: 1.2, hd: 12 },
    { x:  54, z: -42, hw: 1.2, hd: 14 },
  ];
})();

export function buildArena() {
  const group = new THREE.Group();

  const ground = new THREE.Mesh(
    new THREE.PlaneGeometry(ARENA, ARENA),
    new THREE.MeshStandardMaterial({ color: 0x05070d, metalness: 0.9, roughness: 0.25 }));
  ground.rotation.x = -Math.PI / 2; group.add(ground);

  const grid = new THREE.GridHelper(ARENA, 40, NEON, 0x0a2a33);
  grid.position.y = 0.02;
  grid.material.transparent = true; grid.material.opacity = 0.5; group.add(grid);

  const wallMat = new THREE.MeshStandardMaterial({
    color: 0x0a0d16, metalness: 0.8, roughness: 0.3,
    emissive: new THREE.Color(NEON).multiplyScalar(0.06) });
  for (const w of WALLS) {
    const geo = new THREE.BoxGeometry(w.hw * 2, WALL_H, w.hd * 2);
    const mesh = new THREE.Mesh(geo, wallMat);
    mesh.position.set(w.x, WALL_H / 2, w.z);
    mesh.add(new THREE.LineSegments(new THREE.EdgesGeometry(geo),
      new THREE.LineBasicMaterial({ color: NEON })));
    group.add(mesh);
  }
  return { group, walls: WALLS };
}
```

```python
# tools/glb_io.py
import struct, json
import numpy as np

CT_SIZE = {5120:1,5121:1,5122:2,5123:2,5125:4,5126:4}
CT_NP   = {5120:np.int8,5121:np.uint8,5122:np.int16,5123:np.uint16,5125:np.uint32,5126:np.float32}
NCOMP   = {"SCALAR":1,"VEC2":2,"VEC3":3,"VEC4":4,"MAT4":16}

class GLB:
    def __init__(self, json_dict, bin_bytes):
        self.json = json_dict
        self.bin = bin_bytes

def load_glb(path):
    with open(path, "rb") as f:
        data = f.read()
    magic, ver, length = struct.unpack_from("<III", data, 0)
    assert magic == 0x46546C67, "not a glb"
    off = 12; jc = bc = None
    while off < length:
        clen, ctype = struct.unpack_from("<II", data, off); off += 8
        chunk = data[off:off+clen]; off += clen
        if ctype == 0x4E4F534A: jc = chunk
        elif ctype == 0x004E4942: bc = chunk
    return GLB(json.loads(jc.decode("utf-8")), bc or b"")

def read_accessor(g, idx):
    acc = g.json["accessors"][idx]
    bv = g.json["bufferViews"][acc["bufferView"]]
    comp = acc["componentType"]; nc = NCOMP[acc["type"]]; count = acc["count"]
    dtype = CT_NP[comp]; csize = CT_SIZE[comp]
    base = bv.get("byteOffset",0) + acc.get("byteOffset",0)
    stride = bv.get("byteStride", csize*nc)
    if stride == csize*nc:
        flat = np.frombuffer(g.bin, dtype=dtype, count=count*nc, offset=base)
        return flat.reshape(count, nc).copy()
    out = np.empty((count, nc), dtype=dtype)
    for i in range(count):
        out[i] = np.frombuffer(g.bin, dtype=dtype, count=nc, offset=base+i*stride)
    return out

def _pad4(b: bytearray):
    while len(b) % 4: b.append(0)

class BinWriter:
    """Accumulates accessors+bufferViews into a single packed buffer."""
    def __init__(self):
        self.bin = bytearray(); self.bufferViews = []; self.accessors = []
    def add_accessor(self, arr, type_str, comp_type, target=None, normalized=False):
        arr = np.ascontiguousarray(arr, dtype=CT_NP[comp_type])
        _pad4(self.bin)
        offset = len(self.bin)
        self.bin += arr.tobytes()
        bv = {"buffer":0, "byteOffset":offset, "byteLength":arr.nbytes}
        if target: bv["target"] = target
        bv_idx = len(self.bufferViews); self.bufferViews.append(bv)
        count = arr.shape[0]
        acc = {"bufferView":bv_idx, "componentType":comp_type,
               "count":count, "type":type_str}
        # min/max required for POSITION; harmless elsewhere for VEC3 floats
        if type_str in ("VEC3","VEC2","SCALAR") and comp_type == 5126:
            acc["min"] = arr.min(axis=0).astype(float).tolist()
            acc["max"] = arr.max(axis=0).astype(float).tolist()
        if normalized: acc["normalized"] = True
        acc_idx = len(self.accessors); self.accessors.append(acc)
        return acc_idx

def write_glb(path, json_dict, bin_bytes):
    jb = json.dumps(json_dict, separators=(",",":")).encode("utf-8")
    while len(jb) % 4: jb += b" "
    bb = bytearray(bin_bytes)
    while len(bb) % 4: bb.append(0)
    total = 12 + 8 + len(jb) + 8 + len(bb)
    with open(path, "wb") as f:
        f.write(struct.pack("<III", 0x46546C67, 2, total))
        f.write(struct.pack("<II", len(jb), 0x4E4F534A)); f.write(jb)
        f.write(struct.pack("<II", len(bb), 0x004E4942)); f.write(bb)
```

```python
# tools/render_util.py  — minimal dependency-free point-splat renderer
import numpy as np, struct, zlib

def _png(path, rgb):  # rgb: (H,W,3) uint8
    h, w, _ = rgb.shape
    raw = bytearray()
    for y in range(h):
        raw.append(0); raw += rgb[y].tobytes()
    def chunk(typ, data):
        c = struct.pack(">I", len(data)) + typ + data
        return c + struct.pack(">I", zlib.crc32(typ + data) & 0xffffffff)
    sig = b"\x89PNG\r\n\x1a\n"
    ihdr = struct.pack(">IIBBBBB", w, h, 8, 2, 0, 0, 0)
    with open(path, "wb") as f:
        f.write(sig); f.write(chunk(b"IHDR", ihdr))
        f.write(chunk(b"IDAT", zlib.compress(bytes(raw), 6)))
        f.write(chunk(b"IEND", b""))

def render(pos, normals, path, view="side", W=900, H=600, bg=30):
    """Orthographic point splat shaded by normal·light. view: side|top|front|iso."""
    p = pos.astype(np.float64).copy()
    n = normals.astype(np.float64).copy()
    if view == "side":   # look along +Z; screen x=X, y=Y
        sx, sy, depth = p[:,0], p[:,1], p[:,2]; nl = n
    elif view == "top":  # look along -Y; screen x=X, y=Z
        sx, sy, depth = p[:,0], p[:,2], -p[:,1]; nl = n[:,[0,2,1]]
    elif view == "front":# look along +X; screen x=Z, y=Y
        sx, sy, depth = p[:,2], p[:,1], p[:,0]; nl = n[:,[2,1,0]]
    else:                # iso
        a = np.radians(35); b = np.radians(25)
        sx = p[:,0]*np.cos(a) - p[:,2]*np.sin(a)
        sy = p[:,1]*np.cos(b) + (p[:,0]*np.sin(a)+p[:,2]*np.cos(a))*np.sin(b)
        depth = -(p[:,0]*np.sin(a) + p[:,2]*np.cos(a))
        nl = n
    light = np.array([0.4,0.7,0.6]); light/=np.linalg.norm(light)
    shade = np.clip(nl @ light, 0.1, 1.0)
    # map to screen
    pad = 0.06
    def mp(v): 
        lo,hi=v.min(),v.max(); return (v-lo)/(hi-lo+1e-9)
    xi = (pad+ (1-2*pad)*mp(sx))*(W-1)
    yi = ((pad+ (1-2*pad)*(1-mp(sy)))*(H-1))
    xi=xi.astype(int); yi=yi.astype(int)
    img = np.full((H,W,3), bg, np.uint8)
    zbuf = np.full((H,W), -1e9)
    order = np.argsort(depth)  # back to front
    for i in order:
        x,y = xi[i], yi[i]
        if 0<=x<W and 0<=y<H and depth[i] > zbuf[y,x]:
            zbuf[y,x]=depth[i]
            c = int(40+200*shade[i]); img[y,x]=(c,c,c)
    _png(path, img)
    return path
```

```python
# tools/segment.py
import numpy as np
from dataclasses import dataclass

HULL, TURRET, BARREL, WHEELS_L, WHEELS_R, BELT_L, BELT_R = range(7)
PART_NAMES = {HULL:"Hull",TURRET:"Turret",BARREL:"Barrel",
              WHEELS_L:"WheelsLeft",WHEELS_R:"WheelsRight",
              BELT_L:"BeltLeft",BELT_R:"BeltRight"}

@dataclass
class Config:
    front_sign: int = 1       # +1 if barrel points +X (from Task 3)
    belt_z_min: float = 0.40  # |Z| >= belt outer shell (Task 3)
    band_y_max: float = 0.04  # running gear sits below this Y
    side_z_min: float = 0.20  # |Z| >= this counts as a side (not center hull)
    turret_y_min: float = 0.15
    barrel_z_max: float = 0.10
    barrel_x_min: float = 0.55  # barrel reaches beyond this |X| at turret height

def side_of(z, thr=0.20):
    return 1 if z > thr else (-1 if z < -thr else 0)

def classify(pos, idx, cfg: Config):
    """Return per-triangle label by centroid."""
    cen = pos[idx].mean(axis=1)
    X,Y,Z = cen[:,0],cen[:,1],cen[:,2]
    n = len(cen)
    lab = np.full(n, HULL, dtype=np.int32)

    # Turret: upper-central mass
    turret = (Y > cfg.turret_y_min) & (np.abs(Z) < cfg.side_z_min + 0.05)
    lab[turret] = TURRET
    # Barrel: thin, high, reaching toward the front
    barrel = turret & (np.abs(Z) < cfg.barrel_z_max) & \
             (cfg.front_sign * X > cfg.barrel_x_min)
    lab[barrel] = BARREL

    # Running gear band (below body, on a side)
    band = (Y < cfg.band_y_max) & (np.abs(Z) > cfg.side_z_min)
    belt = band & (np.abs(Z) >= cfg.belt_z_min)
    wheel = band & (np.abs(Z) < cfg.belt_z_min)
    lab[belt  & (Z>0)] = BELT_L
    lab[belt  & (Z<0)] = BELT_R
    lab[wheel & (Z>0)] = WHEELS_L
    lab[wheel & (Z<0)] = WHEELS_R
    return lab

def part_indices(labels, part):
    """Triangle indices (rows) for a part."""
    return np.nonzero(labels == part)[0]
```

```python
# tools/beltuv.py
import numpy as np

def arc_length_uv(pos, center_xy, repeats, z_min, z_max):
    """U: angle about (cx,cy) in X-Y -> [0,repeats). V: across width via Z -> [0,1].
    pos: (N,3). Returns (N,2) float32."""
    cx, cy = center_xy
    dx = pos[:,0] - cx
    dy = pos[:,1] - cy
    ang = np.arctan2(dy, dx)          # [-pi, pi]
    frac = (ang + np.pi) / (2*np.pi)  # [0,1)
    u = frac * repeats
    az = np.abs(pos[:,2])
    v = np.clip((az - z_min) / (z_max - z_min + 1e-9), 0.0, 1.0)
    return np.column_stack([u, v]).astype(np.float32)

def loop_center(belt_pos):
    """Reasonable loop center for a side: median X, mid Y of the belt verts."""
    cx = float(np.median(belt_pos[:,0]))
    cy = float((belt_pos[:,1].min() + belt_pos[:,1].max()) * 0.5)
    return (cx, cy)
```

```js
// game/core.js  — pure logic, no Three.js
export class DriveModel {
  constructor({ maxSpeed = 12, accel = 14, decel = 20, turnRate = 1.6,
                trackHalfWidth = 0.45 } = {}) {
    Object.assign(this, { maxSpeed, accel, decel, turnRate, trackHalfWidth });
    this.speed = 0; this.heading = 0; this.throttle = 0; this.steer = 0;
    this.vLeft = 0; this.vRight = 0;
  }
  setInput({ throttle = 0, steer = 0 }) {
    this.throttle = Math.max(-1, Math.min(1, throttle));
    this.steer = Math.max(-1, Math.min(1, steer));
  }
  update(dt) {
    // longitudinal speed toward throttle target
    const target = this.throttle * this.maxSpeed;
    const rate = (Math.abs(target) > Math.abs(this.speed)) ? this.accel : this.decel;
    if (this.speed < target) this.speed = Math.min(target, this.speed + rate * dt);
    else                     this.speed = Math.max(target, this.speed - rate * dt);
    // turning: scaled by steer; allow pivot turns at low speed
    const turn = this.steer * this.turnRate *
                 (0.4 + 0.6 * Math.min(1, Math.abs(this.speed) / this.maxSpeed + 0.4));
    this.heading += turn * dt;
    // per-side linear track speed = body speed ± yaw*halfWidth
    const yawSpeed = turn * this.trackHalfWidth;
    this.vLeft  = this.speed - yawSpeed;
    this.vRight = this.speed + yawSpeed;
    const dx = Math.sin(this.heading) * this.speed * dt;
    const dz = Math.cos(this.heading) * this.speed * dt;
    return { dx, dz, speed: this.speed, heading: this.heading,
             vLeft: this.vLeft, vRight: this.vRight };
  }
}

export function aimYaw(current, target, rate, dt) {
  let diff = target - current;
  while (diff >  Math.PI) diff -= 2 * Math.PI;
  while (diff < -Math.PI) diff += 2 * Math.PI;
  const step = rate * dt;
  if (Math.abs(diff) <= step) return target;
  return current + Math.sign(diff) * step;
}

export class Weapon {
  constructor({ cooldown = 0.8 } = {}) { this.cooldown = cooldown; this.last = -1e9; }
  tryFire(now) {
    if (now - this.last >= this.cooldown) { this.last = now; return true; }
    return false;
  }
}

export function stepShell(sh, dt) {
  sh.pos.x += sh.vel.x * dt; sh.pos.y += sh.vel.y * dt; sh.pos.z += sh.vel.z * dt;
  sh.life -= dt;
}
export function hitsSphere(p, c, r) {
  const dx = p.x - c.x, dy = p.y - c.y, dz = p.z - c.z;
  return dx*dx + dy*dy + dz*dz <= r*r;
}
```

```js
import * as THREE from "three";
import { GLTFLoader } from "three/addons/loaders/GLTFLoader.js";

export function makeTreadTexture() {
  const c = document.createElement("canvas"); c.width = 64; c.height = 256;
  const g = c.getContext("2d");
  g.fillStyle = "#15161a"; g.fillRect(0, 0, 64, 256);
  for (let y = 0; y < 256; y += 32) {            // tread cleats across width
    g.fillStyle = "#2b2d33"; g.fillRect(0, y + 4, 64, 20);
    g.fillStyle = "#0c0d10"; g.fillRect(0, y + 24, 64, 8);
  }
  const tex = new THREE.CanvasTexture(c);
  tex.wrapS = tex.wrapT = THREE.RepeatWrapping;
  tex.repeat.set(1, 1);
  return tex;
}

export async function loadTank(url) {
  const gltf = await new GLTFLoader().loadAsync(url);
  const root = gltf.scene;
  const byName = (n) => root.getObjectByName(n) || console.warn("MISSING part:", n) || null;
  const parts = {
    hull: byName("Hull"), turret: byName("Turret"), barrel: byName("Barrel"),
    wheelsLeft: byName("WheelsLeft"), wheelsRight: byName("WheelsRight"),
    beltLeft: byName("BeltLeft"), beltRight: byName("BeltRight"),
  };
  // Assign tread texture to belt materials; keep two textures so L/R scroll independently
  const treadL = makeTreadTexture(), treadR = makeTreadTexture();
  for (const [obj, tex] of [[parts.beltLeft, treadL], [parts.beltRight, treadR]]) {
    obj?.traverse((m) => {
      if (m.isMesh && m.material) {
        m.material = new THREE.MeshStandardMaterial({
          map: tex, roughness: 0.95, metalness: 0.0, side: THREE.DoubleSide });
      }
    });
  }
  return { root, parts, treadL, treadR };
}

// Per-frame: spin wheels around their Z axle, scroll belt tread, by side speed.
export function applyTrackAnim(t, vLeft, vRight, dt,
                              { wheelK = 6.0, beltK = 1.5 } = {}) {
  if (t.parts.wheelsLeft)  t.parts.wheelsLeft.rotation.z  -= vLeft  * dt * wheelK;
  if (t.parts.wheelsRight) t.parts.wheelsRight.rotation.z -= vRight * dt * wheelK;
  t.treadL.offset.y -= vLeft  * dt * beltK;   // V-scroll = along the loop (U=width here)
  t.treadR.offset.y -= vRight * dt * beltK;
}

export function setTurretYaw(t, yaw) {
  if (t.parts.turret) t.parts.turret.rotation.y = yaw;
  if (t.parts.barrel) t.parts.barrel.rotation.y = yaw;
}

export function barrelTip(t) {
  // world position a bit in front of the turret pivot, along its yaw
  const p = new THREE.Vector3();
  (t.parts.barrel || t.parts.turret || t.root).getWorldPosition(p);
  return p;
}
```

```html
  for(let i=shells.length-1;i>=0;i--){
    const sh=shells[i]; stepShell(sh,dt); sh.mesh.position.set(sh.pos.x,sh.pos.y,sh.pos.z);
    let dead = sh.life<=0;
    for(const tg of targets){ if(tg.visible &&
        hitsSphere(sh.pos,{x:tg.position.x,y:tg.position.y,z:tg.position.z},tg.userData.radius)){
      tg.visible=false; dead=true; boom(); break; } }
    if(dead){ scene.remove(sh.mesh); shells.splice(i,1); }
  }

  // third-person chase camera (behind hull, elevated)
  const back = new THREE.Vector3(-Math.sin(s.heading), 0, -Math.cos(s.heading));
  const camPos = tankGroup.position.clone()
    .add(back.multiplyScalar(SCALE*2.4)).add(new THREE.Vector3(0,SCALE*1.6,0));
  camera.position.lerp(camPos, 0.12);
  camera.lookAt(tankGroup.position.x, SCALE*0.4, tankGroup.position.z);

  document.getElementById("reload").textContent =
    (performance.now()/1000 - weapon.last) < weapon.cooldown ? "…laden" : "";
  renderer.render(scene, camera);
}
loop();
</script></body></html>
```
